        from typing import Tuple, Listimport math# Используем Axial Coordinates (q, r)# https://www.redblobgames.com/grids/hexagons/def hex_distance(a: Tuple[int, int], b: Tuple[int, int]) -> int:    """Расстояние между двумя гексами"""    q1, r1 = a    q2, r2 = b    return (abs(q1 - q2) + abs(q1 + r1 - q2 - r2) + abs(r1 - r2)) // 2def hex_neighbors(q: int, r: int) -> List[Tuple[int, int]]:    """6 соседей гекса"""    directions = [        (1, 0), (1, -1), (0, -1),        (-1, 0), (-1, 1), (0, 1)    ]    return [(q + dq, r + dr) for dq, dr in directions]def hex_line(start: Tuple[int, int], end: Tuple[int, int]) -> List[Tuple[int, int]]:    """Линия гексов от start до end"""    q1, r1 = start    q2, r2 = end    N = hex_distance(start, end)        results = []    for i in range(N + 1):        t = i / N if N > 0 else 0        q = q1 + (q2 - q1) * t        r = r1 + (r2 - r1) * t        results.append(hex_round(q, r))        return resultsdef hex_round(q: float, r: float) -> Tuple[int, int]:    """Округление дробных координат"""    s = -q - r        rq = round(q)    rr = round(r)    rs = round(s)        q_diff = abs(rq - q)    r_diff = abs(rr - r)    s_diff = abs(rs - s)        if q_diff > r_diff and q_diff > s_diff:        rq = -rr - rs    elif r_diff > s_diff:        rr = -rq - rs        return (rq, rr)def hex_to_pixel(q: int, r: int, size: float = 10.0) -> Tuple[float, float]:    """Конвертация hex → pixel (flat-top orientation)"""    x = size * (3/2 * q)    y = size * (math.sqrt(3)/2 * q + math.sqrt(3) * r)    return (x, y)def pixel_to_hex(x: float, y: float, size: float = 10.0) -> Tuple[int, int]:    """Конвертация pixel → hex"""    q = (2/3 * x) / size    r = (-1/3 * x + math.sqrt(3)/3 * y) / size    return hex_round(q, r)
